2. Canvas

SVG Canvas




Code: 

<div id="canvas">
  <h1>SVG Canvas</h1><br><br><br>

  <canvas
    id="my-canvas-text"
    width="200"
    height="100"
    style="border: 1px solid red"
  ></canvas>
  <script>
    var ctx = document
      .getElementById("my-canvas-text")
      .getContext("2d");
    ctx.font = "20px Arial";
    ctx.fillStyle = "blue";
    ctx.fillText("Sheikh Ahasunul", 10, 50);
  </script>

  <canvas
    id="my-canvas-line"
    width="200"
    height="100"
    style="border: 1px solid red"
  ></canvas>
  <script>
    var ctx = document
      .getElementById("my-canvas-line")
      .getContext("2d");
    ctx.moveTo(0, 0);
    ctx.lineTo(200, 100);
    ctx.strokeStyle = "red";
    ctx.lineWidth = 5;
    ctx.stroke();
  </script>

  <canvas
    id="my-canvas-circle"
    width="200"
    height="100"
    style="border: 1px solid red"
  ></canvas>
  <script>
    var ctx = document
      .getElementById("my-canvas-circle")
      .getContext("2d");
    ctx.beginPath();
    ctx.arc(100, 50, 40, 0, 2 * Math.PI);
    ctx.strokeStyle = "red";
    ctx.lineWidth = 4;
    ctx.stroke();
  </script>

  <canvas
    id="my-canvas-rect"
    width="200"
    height="100"
    style="border: 1px solid red"
  ></canvas>
  <script>
    var ctx = document
      .getElementById("my-canvas-rect")
      .getContext("2d");
    ctx.rect(20, 20, 160, 60);
    ctx.strokeStyle = "red";
    ctx.lineWidth = 3;
    ctx.stroke();
  </script>

  <canvas
    id="my-canvas-gradient"
    width="200"
    height="100"
    style="border: 1px solid red"
  ></canvas>
  <script>
    var ctx = document
      .getElementById("my-canvas-gradient")
      .getContext("2d");
    var grd = ctx.createLinearGradient(0, 0, 200, 0);
    grd.addColorStop(0, "red");
    grd.addColorStop(1, "yellow");
    ctx.fillStyle = grd;
    ctx.fillRect(20, 20, 160, 60);
  </script>

  <canvas
    id="my-canvas-image"
    width="200"
    height="100"
    style="border: 1px solid red"
  ></canvas>
  <script>
    var ctx = document
      .getElementById("my-canvas-image")
      .getContext("2d");
    const image = new Image();
    image.onload = function () {
      ctx.drawImage(image, 30, 0, 150, 150);
    };
    image.src = "/Image/envlope.jpg";
  </script>
</div>
